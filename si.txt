package Actions;

import java.util.NoSuchElementException;

import org.openqa.selenium.By;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.PageFactory;
import org.openqa.selenium.support.pagefactory.AjaxElementLocatorFactory;

import Controllers.Controller_Breb_B;
import Locators.Locators_Registros_Breb_b;
import library.common.Util;
import library.core.BasePageWeb;
import library.reporting.Reporter;

public class Page_Registro_Breb_b extends BasePageWeb {

    // ── Constantes ─────────────────────────────────────────────────────────────
    private static final int MAX_INTENTOS          = 15;
    private static final int MAX_INTENTOS_SHADOW   = 10;
    private static final int TIMEOUT_ELEMENT       = 10;
    private static final int TIMEOUT_REGISTRO      = 5;
    private static final long SLEEP_MS             = 500L;

    private static final String SHADOW_SCRIPT      = "return arguments[0].shadowRoot.querySelector(arguments[1])";
    private static final String SHADOW_HOST_CSS    = "dav-back-button";
    private static final String SHADOW_LABEL_CSS   = "label.dav-backButton__label";
    private static final String MSG_NO_RESULTADOS  = "no tiene resultados";

    // ── Campos ─────────────────────────────────────────────────────────────────
    private final Locators_Registros_Breb_b loc;

    @SuppressWarnings("unused")
    private final WebElement webIfrmaePlataformPyEM = Controller_Breb_B.cargarIfrmeContencion;
    @SuppressWarnings("unused")
    private final WebElement webIfrmaeEMContenido   = Controller_Breb_B.cargarIfrmeContenido;

    // ── Constructor ────────────────────────────────────────────────────────────
    public Page_Registro_Breb_b(BasePageWeb parentPage) {
        super(parentPage);
        this.loc = initializeLocators();
    }

    // ── Inicialización ─────────────────────────────────────────────────────────
    private Locators_Registros_Breb_b initializeLocators() {
        Locators_Registros_Breb_b locators = new Locators_Registros_Breb_b();
        AjaxElementLocatorFactory factory = new AjaxElementLocatorFactory(this.getDriver(), 0);
        PageFactory.initElements(factory, locators);
        return locators;
    }

    // ── Métodos públicos ───────────────────────────────────────────────────────

    /**
     * Registra una llave si no existe previamente.
     * Si la llave ya existe, registra el fallo y no continúa.
     *
     * @param nombreLlaVinc Nombre de la llave a registrar.
     */
    public void registrarLlave(String nombreLlaVinc) {
        if (!buscarLlaveExistente(nombreLlaVinc)) {
            return;
        }

        waitElement(loc.bottonRegistro, TIMEOUT_REGISTRO);
        click(loc.bottonRegistro);
        Util.wait(3);

        esperarFinCarga();
    }

    /**
     * Interactúa con un elemento dentro de un Shadow DOM.
     *
     * @param shadowHostLocator Locator del host del shadow root.
     * @param cssInsideShadow   Selector CSS del elemento dentro del shadow root.
     * @param action            Acción a ejecutar: "click", "write", "text", "displayed".
     * @param value             Valor usado solo en la acción "write".
     * @return Resultado de la acción (null para click/write, String para text, Boolean para displayed).
     */
    public Object shadowAction(By shadowHostLocator, String cssInsideShadow, String action, String value) {
        WebElement host    = findElementWithRetry(shadowHostLocator);
        WebElement element = findShadowElementWithRetry(host, cssInsideShadow);

        return executeShadowAction(element, action, value);
    }

    // ── Métodos privados ───────────────────────────────────────────────────────

    /**
     * Busca si la llave ya existe en el sistema.
     *
     * @return true si NO existe (flujo puede continuar), false si ya existe.
     */
    private boolean buscarLlaveExistente(String nombreLlaVinc) {
        changeFrame(loc.frameBreb);
        waitElement(loc.searchBarLlav, TIMEOUT_ELEMENT);
        write(loc.searchBarLlav, nombreLlaVinc);
        click(loc.botonSearch);

        String resultadoBusqueda = getText(loc.noRegistro);

        if (resultadoBusqueda.contains(MSG_NO_RESULTADOS)) {
            Reporter.reportEvent(Reporter.MIC_INFO, "No hay una llave registrada, se procede con el flujo.");
            return true;
        }

        Reporter.reportEvent(Reporter.MIC_FAIL, "Ya existe una llave registrada con ese nombre.");
        return false;
    }

    /**
     * Espera a que el loader desaparezca y el botón de retroceso esté visible.
     */
    private void esperarFinCarga() {
        for (int intentos = 0; intentos < MAX_INTENTOS; intentos++) {
            boolean cargando = isDisplayed(loc.sectionLoader);
            boolean backButtonVisible = (boolean) shadowAction(
                By.cssSelector(SHADOW_HOST_CSS),
                SHADOW_LABEL_CSS,
                "displayed",
                null
            );

            if (!cargando && backButtonVisible) {
                break;
            }

            Util.wait(1);
        }
    }

    /**
     * Busca el shadow host con reintentos.
     */
    private WebElement findElementWithRetry(By locator) {
        for (int i = 0; i < MAX_INTENTOS_SHADOW; i++) {
            try {
                WebElement element = getDriver().findElement(locator);
                if (element != null) return element;
            } catch (Exception e) {
                sleep(SLEEP_MS);
            }
        }
        throw new NoSuchElementException("No se encontró el shadow host: " + locator);
    }

    /**
     * Busca un elemento dentro del shadow root con reintentos.
     */
    private WebElement findShadowElementWithRetry(WebElement host, String cssInsideShadow) {
        JavascriptExecutor js = getJsExecutor();

        for (int i = 0; i < MAX_INTENTOS_SHADOW; i++) {
            try {
                WebElement element = (WebElement) js.executeScript(SHADOW_SCRIPT, host, cssInsideShadow);
                if (element != null) return element;
            } catch (Exception e) {
                sleep(SLEEP_MS);
            }
        }
        throw new NoSuchElementException("No se encontró el elemento en el shadow DOM: " + cssInsideShadow);
    }

    /**
     * Ejecuta la acción indicada sobre el elemento del shadow DOM.
     */
    private Object executeShadowAction(WebElement element, String action, String value) {
        JavascriptExecutor js = getJsExecutor();

        switch (action.toLowerCase()) {
            case "click":
                js.executeScript("arguments[0].click();", element);
                return null;

            case "write":
                element.clear();
                element.sendKeys(value);
                return null;

            case "text":
                return js.executeScript("return arguments[0].innerText;", element);

            case "displayed":
                return js.executeScript("return arguments[0].offsetParent !== null;", element);

            default:
                throw new IllegalArgumentException("Acción no soportada: " + action);
        }
    }

    /**
     * Retorna el JavascriptExecutor del driver actual.
     */
    private JavascriptExecutor getJsExecutor() {
        return (JavascriptExecutor) getDriver();
    }

    /**
     * Pausa la ejecución por los milisegundos indicados.
     */
    public void sleep(long millis) {
        try {
            Thread.sleep(millis);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
}
