public String selectOption(By locator, String opcion) throws Exception {
    WebElement selectElement = element(locator);

    if (selectElement == null) {
        throw new Exception("El elemento no fue encontrado: " + locator);
    }

    click(locator);

    List<WebElement> options = selectElement.findElements(By.tagName("option"));

    for (WebElement option : options) {
        String text = option.getText().trim();

        if (text.equalsIgnoreCase(opcion.trim())) {
            option.click();
            return text;
        }
    }

    throw new Exception("No se encontró la opción: " + opcion);
}
