function renombrarCarpetasSecuencial() {
  // ID de la carpeta que contiene las subcarpetas
  var parentFolderId = "TU_ID_DE_CARPETA";
  var parentFolder = DriveApp.getFolderById(parentFolderId);
  var subFolders = parentFolder.getFolders();
  var folders = [];

  // Paso 1: meter todas las subcarpetas en un array
  while (subFolders.hasNext()) {
    folders.push(subFolders.next());
  }

  // (Opcional) ordenar por nombre actual
  folders.sort(function(a, b) {
    return a.getName().localeCompare(b.getName(), 'es', {sensitivity: 'base'});
  });

  // Paso 2: renombrar con números
  for (var i = 0; i < folders.length; i++) {
    var nuevoNombre = (i + 1).toString(); // “1”, “2”, “3”, …
    folders[i].setName(nuevoNombre);
  }
}
