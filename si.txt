public String selectOption(By locator, String opcion, boolean exacto) throws Exception {
    String msg = "";

    if (opcion == null) opcion = "";
    opcion = opcion.trim();

    WebElement list = this.element(locator);
    if (list == null) {
        throw new Exception("Elemento no encontrado: " + locator);
    }

    // Abrir lista
    try {
        this.click(locator);
    } catch (Exception e) {
        list.click();
    }

    List<WebElement> options = list.findElements(By.tagName("option"));

    int i = 0;
    while (i < options.size()) {
        try {
            WebElement seleoption = options.get(i);
            String text = seleoption.getText() == null ? "" : seleoption.getText().trim();

            boolean coincide;
            if (exacto) {
                coincide = text.equalsIgnoreCase(opcion);
            } else {
                coincide = text.toLowerCase().contains(opcion.toLowerCase());
            }

            if (coincide) {
                seleoption.click();
                msg = text;
                Util.wait(1);
                break; // ← sale del while inmediatamente
            }

            i++;

        } catch (StaleElementReferenceException e) {
            // Reintenta si el DOM cambió
            list = this.element(locator);
            options = list.findElements(By.tagName("option"));
            i = 0;
        }
    }

    return msg;
}
