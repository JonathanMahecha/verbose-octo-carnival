package Actions;

import java.awt.AWTException;
import java.util.ArrayList;
import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.PageFactory;
import org.openqa.selenium.support.pagefactory.AjaxElementLocatorFactory;

import com.lowagie.text.pdf.AcroFields.Item;

import Locators.Locators_Consultar_llaves_Breb_b;
import library.common.Util;
import library.core.BasePageWeb;
import library.reporting.Evidence;
import library.reporting.Reporter;

public class Page_Consultar_llaves_Breb_b extends BasePageWeb {

	Locators_Consultar_llaves_Breb_b loc;
	Page_Registro_Breb_b pageRegis;
	BasePageWeb parentPage;
	// ****************************************************************************************************************************

	public Page_Consultar_llaves_Breb_b(BasePageWeb parentPage) {
		super(parentPage);
		this.parentPage = parentPage;
		initializeLocators();
	}

	private void initializeLocators() {
		loc = new Locators_Consultar_llaves_Breb_b();
		AjaxElementLocatorFactory ajaxElementLocatorFactory = new AjaxElementLocatorFactory(this.getDriver(), 0);
		PageFactory.initElements(ajaxElementLocatorFactory, this.loc);
	}

	public void consultarLlave(String tipoBusqu, String nombreLlaVinc, String filtrosBusqu) {
		pageRegis = new Page_Registro_Breb_b(parentPage);

		if (tipoBusqu != null && tipoBusqu.equalsIgnoreCase("Buscar por nombre de llave o Codigo")) {
			if (pageRegis.llaveYaExiste(nombreLlaVinc)) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No existen llaves con esta caracteristica.");
				return;
			}
		}

		buscarLlave(tipoBusqu);

 		Util.wait(1);
	}

	private void buscarLlave(String tipoBusqu) {
		List<WebElement> filas = this
				.findElements(By.xpath("//table[contains(@class,'table')]//tbody//tr[contains(@title,'Clic')]"));

		for (WebElement fila : filas) {
			List<WebElement> columnas = fila.findElements(By.tagName("td"));

			StringBuilder sb = new StringBuilder();
			int limite = Math.min(4, columnas.size());

			for (int i = 0; i < limite; i++) {
				sb.append(columnas.get(i).getText().trim());
				if (i < limite - 1) {
					sb.append(" | ");
				}
			}

			Reporter.reportEvent(Reporter.MIC_INFO, sb.toString());

			if (tipoBusqu != null && tipoBusqu.equalsIgnoreCase("Buscar por nombre de llave o Codigo")) {
				try {
					fila.click();
					waitElement(By.xpath("//*[contains(text(),'Administración de la llave')]"));
					String llave = getText(By.xpath("//span[contains(@class,'value')]"));
					Evidence.saveFullPage(llave, this);
					Util.wait(1);
					click(By.xpath("//button[@class='back-button']"));
					waitElement(element(By.xpath("//input[contains(@placeholder,'Buscar llave')]")), 10);
				} catch (Exception e) {
					Reporter.reportEvent(Reporter.MIC_WARNING, "No se pudo hacer click en la fila.");
				}
			}
		}
	}

}
