		case "CONTENCION":
			return "CONTENCION";
		case "3":
		case "OBSOLESCENCIA":
			return "OBSOLESCENCIA";
		case "4":
		case "ONPREMISE":
			return "ONPREMISE";
		case "5":
		case "POST_NUBE":
			return "POST_NUBE";
		case "6":
		case "CONTENCION_NUBE":
			return "CONTENCION_NUBE";
		case "7":
		case "PROYECTOS_NUBE":
			return "PROYECTOS_NUBE";
		case "8":
		case "PROYECTOS_SO":
			return "PROYECTOS_SO";
		case "9":
		case "MEJORAS":
			return "MEJORAS";
		default:
//			reportFail("Opción de ambiente no válida: " + codigoAmbiente);
			return "";
		}
	}
	
	public String comprobarDataConsultas(String filtros) throws Exception {

		if (!filtros.isEmpty()) {
			boolean encontroFiltroRepetidoLlave = false;
			boolean encontroFiltroRepetidoCuenta = false;
			boolean encontroFiltroRepetidoEstado = false;
			String[] filtrosEncontrados = filtros.split(",");
			for (String filtroRequerido : filtrosEncontrados) {
				switch (filtroRequerido.trim()) {
				case "Llave Davivienda":
				case "Código de establecimiento":
				case "Llave Davivienda documento":
					if (encontroFiltroRepetidoLlave == false) {
						encontroFiltroRepetidoLlave = true;
					} else {
						Reporter.reportEvent(Reporter.MIC_WARNING,
								"Solo se admite un Filtro por categoria: Llaves, Cuentas y Estado");
						SettingsRun.exitTestIteration();
					}
					break;

				case "Ahorros":
				case "Corriente":
					if (encontroFiltroRepetidoCuenta == false) {
						encontroFiltroRepetidoCuenta = true;
					} else {
						Reporter.reportEvent(Reporter.MIC_WARNING,
								"Solo se admite un Filtro por categoria: Llaves, Cuentas y Estado");
						SettingsRun.exitTestIteration();
					}
					break;
				case "Activa":
				case "Bloqueada":
					if (encontroFiltroRepetidoEstado == false) {
						encontroFiltroRepetidoEstado = true;
					} else {
						Reporter.reportEvent(Reporter.MIC_WARNING,
								"Solo se admite un Filtro por categoria: Llaves, Cuentas y Estado");
						SettingsRun.exitTestIteration();
					}
					break;

				default:
					Reporter.reportEvent(Reporter.MIC_WARNING, "No se encuentra ninguna opcion valida");
					SettingsRun.exitTestIteration();
					break;
				}
			}
			return "Se han encontrado filtros";
		} else {
			return "No hay filtros por comprobar";
		}
	}

	public Object seleccionTipoDeVinculacionLlavePersona(String nombreRegistro, String seleccionTipoDeLlave)

					Reporter.reportEvent(Reporter.MIC_PASS,

					return true;
				}
			}
			break;
		// falta agregar todas las opciones para este modulo

			pageRegistro.seleccionTipoDeVinculacionLlave("no");
			return true;

			return null;
		}

	public void ConsultaDeLlavesBrebNewDiseño(List<String> dataExcel) throws Exception {
		Page_Consultar_llaves_Breb_b pageConsulta = new Page_Consultar_llaves_Breb_b(webElemPlataformPyEM);
		switch (dataExcel.get(4)) {
		case "Extraer todas las llaves":
			Object ListasDeLlaves = pageConsulta.ConsultaTodasLasLlaves(dataExcel.get(3), true);
			if (ListasDeLlaves == null) {
				Reporter.reportEvent(Reporter.MIC_INFO, "No se encontro la llave: " + dataExcel.get(3));
				cierreDeSesionMultiPlataforma();
			} else if (ListasDeLlaves.equals("Se genera la informacion de todas las llaves")) {
				Reporter.reportEvent(Reporter.MIC_INFO, "Se ha generado la informacion de la llaves encontradas");
			} else if (ListasDeLlaves.equals("No se encontraron llaves")) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No hay llaves disponibles.");
				cierreDeSesionMultiPlataforma();
			} else {
				Reporter.reportEvent(Reporter.MIC_PASS,
						"***[Se encuentra la Llave Solicitada]*** " + "\n" + ListasDeLlaves);
			}
			break;

		case "Buscar por nombre de llave o Codigo":
			switch (dataExcel.get(8)) {
			case "Consulta Empresa":
				msgError = pageConsulta.consultaLlaveBarraBuscadora(dataExcel.get(5));
				if (msgError.contains("")) {
					if (dataExcel.get(3).isEmpty() && !dataExcel.get(7).isEmpty()) {
						msgInfo = pageConsulta.validarLlaveConsulta(dataExcel.get(7));
					} else if (dataExcel.get(7).isEmpty() && !dataExcel.get(3).isEmpty()) {
						msgInfo = pageConsulta.validarLlaveConsulta("@" + dataExcel.get(3));
					} else if (!dataExcel.get(3).isEmpty() && !dataExcel.get(7).isEmpty()) {
						msgInfo = pageConsulta.validarLlaveConsulta("@" + dataExcel.get(3) + " - " + dataExcel.get(7));
					}
					reporteLlaveEncontrada(msgInfo);
				}
				break;

			case "Consulta Codigo Establecimiento":
				pageConsulta.consultaLlaveBarraBuscadora(dataExcel.get(7));
				if (!dataExcel.get(7).isEmpty()) {
					msgInfo = pageConsulta.validarLlaveConsulta(dataExcel.get(7));
				} else {
					Reporter.reportEvent(Reporter.MIC_INFO, "La casilla se encuentra vacia");
					cierreDeSesionMultiPlataforma();
				}
				reporteLlaveEncontrada(msgInfo);
				break;
			case "Consulta Nombre de Llave":
				pageConsulta.consultaLlaveBarraBuscadora("@" + dataExcel.get(3));
				if (!dataExcel.get(7).isEmpty()) {
					msgInfo = pageConsulta.validarLlaveConsulta("@" + dataExcel.get(3));
				} else {
					Reporter.reportEvent(Reporter.MIC_INFO, "La casilla se encuentra vacia");
					cierreDeSesionMultiPlataforma();
				}
				reporteLlaveEncontrada(msgInfo);
				break;

			default:
				Reporter.reportEvent(Reporter.MIC_INFO, "No se ha seleccionado una opcion valida");
				cierreDeSesionMultiPlataforma();
				break;
			}
			break;

		default:
			Reporter.reportEvent(Reporter.MIC_INFO, "No se ha seleccionado una opcion valida");
			cierreDeSesionMultiPlataforma();
			break;
		}
	}

	public String cierreSesionIteracionDependienteCuenta(boolean login, boolean logout) {

		String msgAlert1 = "los clientes son iguales por lo que se puede continuar en la misma sesion";
		String msgAlert2 = "los clientes no son iguales por lo que no se puede continuar en la misma sesion";
		String msgAlert3 = "Iteration Final";

		boolean iteracionInicial = SettingsRun.esIteracionInicial();
		boolean iteracionFinal = SettingsRun.esIteracionFinal();
		int iteracionActual = SettingsRun.getCurrentIteration();
		int proximaIteracion = SettingsRun.getNextIteration(iteracionActual);
		String numeroCuentaCurrentIterationAnterior = null;
		String numeroCuentaCurrentIteration = SettingsRun.getTestData().getParameterByExec("Id usuario",
				iteracionActual);
		String numeroCuentaNextIteration = SettingsRun.getTestData().getParameterByExec("Id usuario", proximaIteracion);
		if (logout) {
			if (iteracionFinal == true) {
				return msgAlert3 + " " + msgAlert2;
			} else if (iteracionFinal == false) {
				if (numeroCuentaCurrentIteration.equals(numeroCuentaNextIteration)) {
					return msgAlert1;
				} else {
					return msgAlert2;
				}
			}
		} else if (login) {
			if (!iteracionInicial) {
				int iteracionAnterior = SettingsRun.getPreviousIteration();
				numeroCuentaCurrentIterationAnterior = SettingsRun.getTestData().getParameterByExec("Id usuario",
						iteracionAnterior);
			} else if (iteracionInicial) {
				return "se puede hacer login";
			}
			if (iteracionFinal == false) {
				if (numeroCuentaCurrentIterationAnterior.equals(numeroCuentaCurrentIteration)) {
					return msgAlert1;
				} else {
					return msgAlert2;
				}

	public void edicionLlaveSeleccionada(List<String> dataExcel) throws Exception {
		Page_Modificar_Llaves_Bre_b pageModificar = new Page_Modificar_Llaves_Bre_b(webElemPlataformPyEM);
		Page_Eliminar_Llaves_Bre_b pageEliminar = new Page_Eliminar_Llaves_Bre_b(webElemPlataformPyEM);

		}
		pageConsulta.esperaPantallaDeCargaWhite();

	public void flujoSeleccionBlockActiveAandExtension(String tipoDeSeleccion, List<String> dataExcel)
			throws Exception {
		Page_Bloqueo_activacion_Breb_b pageBloActExte = new Page_Bloqueo_activacion_Breb_b(webElemPlataformPyEM);
		Page_Consultar_llaves_Breb_b pageConsulta = new Page_Consultar_llaves_Breb_b(webElemPlataformPyEM);
		Page_Eliminar_Llaves_Bre_b pageEliminar = new Page_Eliminar_Llaves_Bre_b(webElemPlataformPyEM);

		pageConsulta.esperaPantallaDeCargaWhite();
		msgError = pageBloActExte.seleccionDeMenusBlockActiveAandExtension(tipoDeSeleccion);
		if (tipoDeSeleccion.contains("Extension de Bloqueo") && msgError == null) {
			pageBloActExte.seleccionDeMenusBlockActiveAandExtension("Bloqueo de llave");
			pageConsulta.esperaPantallaDeCargaWhite();
			pageBloActExte.generarInfoLLave();
			pageConsulta.esperaPantallaDeCargaWhite();
			msgError = pageEliminar.getTextoYClickEnShadowElement("dav-modal", "h3.dav-modalInfo__title",
					"#modalConfirm button");
			if (!msgError.contains("Bloqueo exitoso")) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No se pudo modifica por lo que no se podra continuar ");
				cierreDeSesionMultiPlataforma();
			} else {
				Reporter.reportEvent(Reporter.MIC_PASS, "la modificacion de la llave fue exitosa");
			}
			pageConsulta.esperaPantallaDeCargaWhite();
			seleccionConsultas();
			consultaLlaveEliminacion(dataExcel);
			seleccionarLlaveCodigoRequeridoEliminacion(dataExcel);
			pageBloActExte.seleccionDeMenusBlockActiveAandExtension(tipoDeSeleccion);
		} else if (msgError == null) {
			if (tipoDeSeleccion.contains("Activacion de llave")) {
				Reporter.reportEvent(Reporter.MIC_WARNING, "ya se encuentra en el estado: Activo");
			} else {
				Reporter.reportEvent(Reporter.MIC_WARNING, "ya se encuentra en el estado: Bloqueado");

			}
			cierreDeSesionMultiPlataforma();
		} else if (msgError.contains("La llave ya se encuentra con extension de bloqueo")) {
			Reporter.reportEvent(Reporter.MIC_WARNING, msgError);
			cierreDeSesionMultiPlataforma();
		}

		pageConsulta.esperaPantallaDeCargaWhite();
		pageBloActExte.generarInfoLLave();
		pageConsulta.esperaPantallaDeCargaWhite();
		msgError = pageEliminar.getTextoYClickEnShadowElement("dav-modal", "h3.dav-modalInfo__title",
				"#modalConfirm button");
		switch (tipoDeSeleccion) {
		case "Bloqueo de llave":
			if (!msgError.contains("Bloqueo exitoso")) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No se pudo modifica por lo que no se podra continuar ");
				cierreDeSesionMultiPlataforma();
			} else {
				Reporter.reportEvent(Reporter.MIC_PASS, "la modificacion de la llave fue exitosa");
			}
			break;
		case "Extension de Bloqueo":
			if (!msgError.contains("Extensión de bloqueo exitoso")) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No se pudo modifica por lo que no se podra continuar ");
				cierreDeSesionMultiPlataforma();
			} else {
				Reporter.reportEvent(Reporter.MIC_PASS, "la modificacion de la llave fue exitosa");
			}
			break;
		case "Activacion de llave":
			if (!msgError.contains("Reactivación exitosa")) {
				Reporter.reportEvent(Reporter.MIC_FAIL, "No se pudo modifica por lo que no se podra continuar ");
				cierreDeSesionMultiPlataforma();
			} else {
				Reporter.reportEvent(Reporter.MIC_PASS, "la modificacion de la llave fue exitosa");
			}
			break;
		}
		pageConsulta.esperaPantallaDeCargaWhite();
	}

			pageFrontEmp.cerrarSesionFrontEmpresarial();

		if (SettingsRun.getTestData().getParameter("Tipo Cliente").contains("Empresarial")) {
			cierreSesionEmpresarial();
		} else {
			cierreSesionFront();
			;
		}
	}

	public void cierreDeSesionFront(String msgError) throws Exception {
		this.loginFrontPyme = new PageLoginPymes1(loginFrontPyme);
		if (msgError.contains("Iteration Final") && msgError
				.contains("los clientes no son iguales por lo que no se puede continuar en la misma sesion")) {
			loginFrontPyme.CerrarSesionFront();
			SettingsRun.exitTest();
		} else if (msgError.equals("los clientes son iguales por lo que se puede continuar en la misma sesion")) {
			SettingsRun.exitTestIteration();
		}
	}
