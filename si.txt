	public void registrarLlave(String nombreLlaVinc) {
		int intentos = 0;
		int maxIntentos = 15;
		boolean llaveRegistro = buscarLlaveExistente(nombreLlaVinc);
		boolean dato = (boolean) shadowAction(By.cssSelector("dav-back-button"), "label.dav-backButton__label", "displayed", null);
		if (llaveRegistro) {
			waitElement(loc.bottonRegistro, 5);
			click(loc.bottonRegistro);
			Util.wait(3);
			while (intentos < maxIntentos) {
				if (isDisplayed(loc.sectionLoader)) {
					Util.wait(1);
				} else if(dato){
					break;
				}
				intentos++;
			}
		} else {

		}
	}

	private boolean buscarLlaveExistente(String nombreLlaVinc) {
		changeFrame(loc.frameBreb);
		waitElement(loc.searchBarLlav, 10);
		write(loc.searchBarLlav, nombreLlaVinc);
		click(loc.botonSearch);
		String llaveRegistro = getText(loc.noRegistro);
		if (llaveRegistro.contains("no tiene resultados")) {
			Reporter.reportEvent(Reporter.MIC_INFO, "No hay una llave registrada, se procede con el flujo.");
			return true;
		}
		Reporter.reportEvent(Reporter.MIC_FAIL, "Ya existe una llave registrada con ese nombre.");
		return false;
	}

	public Object shadowAction(By shadowHostLocator, String cssInsideShadow, String action, String value) {
		JavascriptExecutor js = (JavascriptExecutor) getDriver();

		int intentos = 0;
		int maxIntentos = 10;
		WebElement host = null;
		WebElement element = null;

		while (intentos < maxIntentos) {
			try {
				host = getDriver().findElement(shadowHostLocator);
				if (host != null)
					break;
			} catch (Exception e) {
				sleep(500);
			}
			intentos++;
		}

		if (host == null) {
			throw new NoSuchElementException("No se encontró el shadow host");
		}

		intentos = 0;
		String script = "return arguments[0].shadowRoot.querySelector(arguments[1])";

		while (intentos < maxIntentos) {
			try {
				element = (WebElement) js.executeScript(script, host, cssInsideShadow);
				if (element != null)
					break;
			} catch (Exception e) {
				sleep(500);
			}
			intentos++;
		}

		if (element == null) {
			throw new NoSuchElementException("No se encontró el elemento dentro del shadow DOM");
		}

		switch (action.toLowerCase()) {
		case "click":
			js.executeScript("arguments[0].click();", element);
			return null;

		case "write":
			element.clear();
			element.sendKeys(value);
			return null;

		case "text":
			return js.executeScript("return arguments[0].innerText;", element);

		case "displayed":
			return (Boolean) js.executeScript("return arguments[0].offsetParent !== null;", element);

		default:
			throw new IllegalArgumentException("Acción no soportada: " + action);
		}
	}

	public void sleep(long millis) {
		try {
			Thread.sleep(millis);
		} catch (InterruptedException e) {
			Thread.currentThread().interrupt();
		}
	}
