public HashMap<String, Double> irTablaValores(String segmento) throws Exception {
		HashMap<String, Double> datosCobros = new HashMap<>();
		int OPCION = 6;
		int SEM = 1;
		String str1;

		read("TABLA DE VALORES GENERAL - Tarifas Base");
		sendKey(1);
		ctrlXE();
		read("SELECCION SEGMENTO");
		int contador = 0;
		String textoEncontrado = "";

		if (!segmento.equalsIgnoreCase("Sin Segmento")) {
			do {
				contador++;
				Util.wait(1);

				textoEncontrado = read();
				if (textoEncontrado != null && textoEncontrado.contains(segmento)) {
					Reporter.reportEvent(Reporter.MIC_DONE, "Se encontro el segmento predeterminado: " + segmento);
					break;
				}
				if (contador >= 30) {
					break;
				}

				right(1);
			} while (true);
		}
		ctrlX(1);
		String textoPantalla = null;
		double descrValorCan = 0.0D;
		int numPantalla = 1;
		while (true) {
			Util.wait(2);
			textoPantalla = read();
			saveEvidence("ConsultaMovs_" + numPantalla);
			numPantalla++;
			String[] datos = textoPantalla.split("\n");
			for (int i = 7; i < 19; i++) {
				String idcobro = datos[i].substring(43, 51).trim();
				String descripcion = datos[i].substring(5, 41).trim();
				String valor = datos[i].substring(53, 68).replace(",", "").trim();
				String cantidad = datos[i].substring(70, 78).trim();
				if (!valor.isEmpty())
					descrValorCan = Double.parseDouble(valor.replace(",", "."));
				datosCobros.put(idcobro, Double.valueOf(descrValorCan));
			}
			ctrlX(3);
			if (textoPantalla.contains("NO HAY MAS PAGINAS")) {
				escapeQ(2);
				return datosCobros;
			}
			if (textoPantalla.contains("pag 12 de 12")) {
				escapeQ(2);
				return datosCobros;
			}
		}
	}
