public String aprobarTx(int numRetosFalla, String valCodSeg, boolean desdeDetalle) throws Exception {

    String msgResp = null;
    boolean isCliVirtual = DatosEmpresarial.TIPO_TOKEN.equals(DatosEmpresarial.TOKEN_OTP);

    NotificacionSMS.loadHour(); // SIRVE PARA OTP
    Util.wait(4);

    if (this.existDialog()) {
        this.acceptDialog();
    }

    Util.wait(4);

    String textnbTaprobar = obtenerTextoBotonAprobar();
    String tipoPrueba = SettingsRun.getTestData().getParameter("Tipo prueba").trim();
    boolean esModuloInternacional = this.element(nbButtonAprointernacional) != null;

    if (esTipoPruebaInscripcion(tipoPrueba)) {
        if (this.element(btnAprobarModificacion) != null) {
            textnbTaprobar = this.getText(this.btnAprobarModificacion);
        }

        if (this.element(btnAprobarEliminacion) != null) {
            textnbTaprobar = this.getText(this.btnAprobarEliminacion);
        }
    }

    if (this.nbButtonAprobarIns.equals(textnbTaprobar)) {
        Util.wait(2);
        this.click(this.btnAprobarInscripcion);
    } else if (this.nbButtonAprobarMod.equals(textnbTaprobar)) {
        Util.wait(2);
        this.click(this.btnAprobarModificacion);
    } else if (this.nbButtonAprobarElm.equals(textnbTaprobar)) {
        Util.wait(2);
        this.click(this.btnAprobarEliminacion);
    } else if (this.nbButtonAprobar.equals(textnbTaprobar)) {
        Util.wait(4);
        this.clickButton(this.nbButtonAprobar);
        Util.wait(5);

        if (this.existDialog()) {
            this.acceptDialog();
        }

        if (this.element(By.id("lblMasterAlerta")) != null) {
            msgResp = this.getMsgAlertIfExist("lblMasterAlerta");
            Evidence.saveAllScreens(msgResp, this);
        }

        if (msgResp != null && !esMensajeExitoso(msgResp)) {
            return msgResp;
        }
    } else if (this.nbButtonAprobarT2.equals(textnbTaprobar)) {
        if (this.element(nbButtonAprointernacional) != null && this.isEnabled(nbButtonAprointernacional)) {
            Util.wait(4);
            this.click(nbButtonAprointernacional);
            Util.wait(4);
        } else {
            Util.wait(5);
            if (this.existDialog()) {
                this.acceptDialog();
            }
            Util.wait(5);
            this.clickButton(this.nbButtonAprobarTxDT);
        }
    } else if (this.nbButtonAprobarT.equals(textnbTaprobar)) {
        if (this.isDisplayed(nbButtonAprointernacional) && this.isEnabled(nbButtonAprointernacional)) {
            Util.wait(6);
            this.click(nbButtonAprointernacional);
            Util.wait(4);
        } else {
            Util.wait(5);
            this.clickButton(this.nbButtonAprobarT);
        }
    }

    if (this.isDisplayed(btnSeguroAprobar)) {
        this.click(btnSeguroAprobar);
    }

    WebElement objToken = null;
    boolean estaHabilitado = false;
    int contador = 0;

    Util.wait(3);

    if (!esModuloInternacional) {
        while (msgResp == null && !estaHabilitado && contador < 60) {
            Util.wait(5);
            contador++;

            msgResp = obtenerMensajeAlerta();
            objToken = buscarTokenNormal();

            if (msgResp != null && msgResp.contains("recibir� un código")) {
                By botonModal = By.xpath("//*[@id='botonModal']");
                if (this.element(botonModal) != null) {
                    this.click(botonModal);
                }
            }

            if (objToken != null) {
                estaHabilitado = this.isEnabled(objToken);
            }
        }
    } else {
        while (msgResp == null && !estaHabilitado && contador < 60) {
            Util.wait(6);
            contador++;

            if (DatosEmpresarial.TIPO_TOKEN.equals(DatosEmpresarial.TOKEN_OTP)) {
                msgResp = this.closeActiveIntAlert();
            }

            String servicio = SettingsRun.getTestData().getParameter("Servicio").trim();
            String tipoPruebatx = SettingsRun.getTestData().getParameter("Tipo prueba").trim();

            if ((servicio.equals("Tx Internacionales Enviar al exterior")
                    || servicio.equals("Tx Internacionales Recibir desde el exterior"))
                    && !tipoPruebatx.equals("Tx Pend aprobación")) {
                msgResp = this.closeActiveIntAlert();
            }

            if (msgResp != null && msgResp.contains("no podemos atender su solicitud")) {
                return msgResp;
            }

            objToken = buscarTokenInternacional();

            if (objToken != null) {
                estaHabilitado = this.isEnabled(objToken);
            } else {
                return "No se habilito el campo del código de Confirmación";
            }
        }
    }

    if (estaHabilitado) {
        if (isCliVirtual) {
            this.moveUpScreen();
            Evidence.saveAllScreens("MsgEnvíoOtp", this);
        }

        Util.wait(1);
        msgResp = this.ingresarDinamica(numRetosFalla);

    } else if (DatosDavivienda.IS_RISKMOTOR) {
        this.fechaHoraTx = new Date();
    } else {
        return "No se habilito el campo del código de Confirmación";
    }

    Evidence.save("AprobarTxh", this);
    Evidence.saveAllScreens("AprobarTx", this);

    return msgResp;
}

private String obtenerTextoBotonAprobar() throws Exception {
    By[] botones = {
            nbButtonApro,
            nbButtonApro1,
            nbButtonApro2,
            nbButtonAprTra,
            nbButtonAprointernacional,
            btnAprobarInscripcion
    };

    for (By btn : botones) {
        if (this.element(btn) != null) {
            Util.wait(1);
            return this.getText(btn);
        }
    }

    return null;
}

private boolean esTipoPruebaInscripcion(String tipoPrueba) {
    return "Inscribir cuenta".equals(tipoPrueba)
            || "Inscripción por archivo".equals(tipoPrueba)
            || "Modificar cuenta".equals(tipoPrueba)
            || "Eliminar cuenta".equals(tipoPrueba)
            || "Consultar Inscripción".equals(tipoPrueba)
            || "Autogestion de servicios".equals(tipoPrueba);
}

private boolean esMensajeExitoso(String msgResp) {
    return msgResp.contains(PageConfirmacion1.MSG_EXITO_PAGO)
            || msgResp.contains(PageConfirmacion1.MSG_EXITO_GUARD)
            || msgResp.contains(PageConfirmacion1.MSG_EXITO_ENV_OTP)
            || msgResp.contains(PageConfirmacion1.MSG_EXITO_APROB);
}

private String obtenerMensajeAlerta() throws Exception {
    String msgResp = this.getMsgAlertIfExist("lblMasterAlerta");
    if (msgResp == null) {
        msgResp = this.getMsgAlertIfExist("AlertaModal");
    }
    if (msgResp == null) {
        msgResp = this.getMsgAlertIfExist("mensaje");
    }
    return msgResp;
}

private WebElement buscarTokenNormal() throws Exception {
    By[] tokens = {
            locCmTokenOtp,
            locCmTokenOtpModif,
            locCmTokenOtpElim,
            locCmTokenOtpPen
    };

    for (By token : tokens) {
        if (this.element(token) != null) {
            Util.wait(3);
            return this.element(token);
        }
    }

    return null;
}

private WebElement buscarTokenInternacional() throws Exception {
    if (this.element(locCmTokenOtpInternacional) != null) {
        Util.wait(3);
        return this.element(locCmTokenOtpInternacional);
    }
    return null;
}
