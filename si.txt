import java.io.IOException;
import java.nio.file.*;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;

private void renombrarConFecha(String rutaActual) throws IOException {
    Path origen = Paths.get(rutaActual);

    // Formato AAAAMMDD
    String fecha = LocalDate.now().format(DateTimeFormatter.BASIC_ISO_DATE);

    String nuevoNombre = "ACIERTA_PN_" + fecha + ".txt";

    Path destino = origen.resolveSibling(nuevoNombre);

    Files.move(
        origen,
        destino,
        StandardCopyOption.REPLACE_EXISTING
    );
}
