private boolean cambiarAVentanaEmergente() throws Exception {

    Util.wait(5);

    List<String> ventanas = new ArrayList<>(loginFrontPyme.getDriver().getWindowHandles());

    if (ventanas.size() < 2) {
        reportFail("No se abrio la segunda pestaña");
        SettingsRun.exitTestIteration();
        return false;
    }

    // Si existen más de 3 ventanas, validar y cerrar la que no contiene el select esperado
    if (ventanas.size() > 3) {
        for (String handle : ventanas) {

            Reporter.reportEvent(Reporter.MIC_INFO, handle);
            loginFrontPyme.getDriver().switchTo().window(handle);

            WebElement empresa = loginFrontPyme.element(
                    By.xpath("//select[@id='dropMasterEmpresa']")
            );

            if (!loginFrontPyme.isDisplayed(empresa)) {
                loginFrontPyme.getDriver().close();
            }

            break; // Se mantiene comportamiento original
        }
    }

    String primeraVentana = ventanas.get(0);
    String segundaVentana = ventanas.get(1);

    Reporter.write("Primera ventana: " + primeraVentana +
                   " || Segunda ventana: " + segundaVentana);

    loginFrontPyme.getDriver().switchTo().window(primeraVentana);
    Util.wait(2);
    loginFrontPyme.getDriver().close();

    loginFrontPyme.getDriver().switchTo().window(segundaVentana);

    return true;
}
