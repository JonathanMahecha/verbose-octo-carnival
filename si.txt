package dav.CobrosMiddle;

import java.util.HashMap;
import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.ui.Select;

import dav.middlePymes.PageInicioMiddle;
import dav.pymes.PageLoginPymes;
import dxc.util.DXCUtil;
import library.reporting.Evidence;
import library.reporting.Reporter;
import library.settings.SettingsRun;

public class PageEmpresaCobros extends PageInicioMiddle {

    private static final int MAX_WAIT_ITERATIONS = 30;
    private static final int WAIT_ONE_SECOND = 1;
    private static final int WAIT_TWO_SECONDS = 2;
    private static final int WAIT_THREE_SECONDS = 3;
    private static final int WAIT_FIVE_SECONDS = 5;

    // Locators
    private final By tablaEmpresaResultados = By.xpath("//*[@id='cphCuerpo_gvEmpresa']");
    private final By tituloAdministracion = By.id("lblMasterTitulo");
    private final By tablaCantAcumulTx = By.xpath("//table[@id='cphCuerpo_gvConteoServicios']");

    private final By inpNumClienteEmpresa = By.id("cphCuerpo_txtBQNumeroClienteEmpresarial");
    private final By inpIdEmpresa = By.id("cphCuerpo_txtBQNumIdentificacionEmpresa");
    private final By btnConsultar = By.id("cphCuerpo_btnQBBuscar");

    private final By labelConteoTx = By.id("cphCuerpo_gvConteoServicios");
    private final By labelSegmento = By.id("cphCuerpo_lblDESegmento");
    private final By labelTotalTxAcumuladas = By.id("cphCuerpo_lblDECantidadTransaccionesAcumuladas");
    private final By labelTxPorEmpresa = By.id("cphCuerpo_lblDECantidadTransaccionesEmpresa");

    private final By btnGrabar = By.id("cphCuerpo_btnGrabar");
    private final By alertError = By.id("lblMasterAlerta");

    // variables inicializadas
    private int[] contadorActualizado = null;

    public PageEmpresaCobros(PageLoginPymes parentPage) {
        super(parentPage);
    }

    private WebElement esperarElemento(By locator, String mensajeTimeout) throws Exception {
        for (int intento = 1; intento <= MAX_WAIT_ITERATIONS; intento++) {
            WebElement elemento = this.element(locator);
            if (elemento != null) {
                return elemento;
            }
            DXCUtil.wait(WAIT_ONE_SECOND);
        }
        Reporter.reportEvent(Reporter.MIC_FAIL, mensajeTimeout);
        SettingsRun.exitTestIteration();
        return null;
    }

    /**
     * Consulta de la empresa
     */
    public void consultarEmpresa() throws Exception {
        this.irAOpcion(null, "Administración", "Empresa");
        Evidence.save("Evidencia Final", this);

        String clienteEmpresa = SettingsRun.getTestData().getParameter("Cliente Empresarial");
        String nitEmpresa = SettingsRun.getTestData().getParameter("Numero ID Empresa");

        WebElement numClienteEmp = esperarElemento(
                inpNumClienteEmpresa,
                "TimeOut"
        );
        Evidence.save("Evidencia empresa", this);

        WebElement numIdEmpresa = this.element(inpIdEmpresa);
        numClienteEmp.sendKeys(clienteEmpresa);
        numIdEmpresa.sendKeys(nitEmpresa);

        WebElement btnBuscar = this.element(btnConsultar);
        btnBuscar.click();
        Evidence.save("Evidencia empresa", this);

        esperarElemento(tablaEmpresaResultados, "TimeOut");

        Evidence.save("Evidencia empresa", this);

        String xpathEditarEmp = "//*[@id='cphCuerpo_gvEmpresa']//td[contains(text(),'" + clienteEmpresa
                + "')]/following-sibling::td[contains(text(),'" + nitEmpresa + "')]/following-sibling::td/a";

        WebElement btnEditar = this.element(By.xpath(xpathEditarEmp));
        DXCUtil.wait(WAIT_FIVE_SECONDS);

        if (btnEditar != null) {
            btnEditar.click();
        } else {
            System.out.println("Se esperaba encontrar resultados que coincidieran con la información.");
            Evidence.save("Evidencia empresa", this);
            SettingsRun.exitTestIteration();
        }

        Evidence.save("Evidencia empresa", this);

        esperarElemento(labelConteoTx, "TimeOut");
    }

    /**
     * Compara información de combo empresa vs administración de combos
     */
    public void comparacionComboEmpresa(
            int eventStatus,
            String msgReport,
            String comboEncontrado,
            String comboInicial,
            List<HashMap<String, String>> results,
            HashMap<String, String> informacion) throws Exception {

        if (comboEncontrado.equals(comboInicial)) {
            eventStatus = Reporter.MIC_PASS;
            msgReport = "El combo '" + comboEncontrado + "' encontrado coinside correctamente con la data entregada.";
            Reporter.reportEvent(eventStatus, msgReport);
        } else {
            eventStatus = Reporter.MIC_WARNING;
            msgReport = "El combo '" + comboEncontrado
                    + "' encontrado no coinside con la data entregada, pero cuenta con combo por lo tanto se consultara y se verificara.";
            Reporter.reportEvent(eventStatus, msgReport);
        }

        comboInicial = comboEncontrado;

        DXCUtil.wait(WAIT_TWO_SECONDS);

        String segmento = this.element(labelSegmento).getText();
        String cantidadTotalTx = this.element(labelTotalTxAcumuladas).getText();

        informacion.put("Cantidad Tx Acumuladas", cantidadTotalTx);
        informacion.put("Segmento", segmento);

        Evidence.saveFullPage("Evidencia empresa", this);
        Reporter.reportEvent(
                Reporter.MIC_INFO,
                "Se ha guardado la siguiente información: [ 'Segmento' : " + segmento + " ]");

        String cantidadMaxTxEmp = this.element(labelTxPorEmpresa).getText();
        this.element(btnGrabar).click();

        esperarElemento(alertError, "TimeOut");

        String cantiMaxTxEmpCombos = informacion.get("Cantidad Maxima Tx Empresa");
        DXCUtil.wait(WAIT_THREE_SECONDS);

        if (Integer.parseInt(cantidadMaxTxEmp) == Integer.parseInt(cantiMaxTxEmpCombos)) {

            String[] destinos = {
                    informacion.get("Destino A cuentas Davivienda"),
                    informacion.get("Destino A cuentas otros bancos"),
                    informacion.get("Destino A Daviplata"),
                    informacion.get("Destino A Depósitos Electrónicos"),
                    informacion.get("Destino A Tarjeta Prepago"),
                    informacion.get("Destino Davivienda + Daviplata")
            };

            String[] cantidadDestino = {
                    informacion.get("A cuentas Davivienda"),
                    informacion.get("A cuentas otros bancos"),
                    informacion.get("A Daviplata"),
                    informacion.get("A Depósitos Electrónicos"),
                    informacion.get("A Tarjeta Prepago"),
                    informacion.get("Davivienda + Daviplata")
            };

            WebElement conteoTxTabla = this.element(tablaCantAcumulTx);
            List<WebElement> filas = conteoTxTabla.findElements(By.tagName("tr"));

            for (int i = 1; i < filas.size() - 1; i++) {
                String xpathComparar = "//*[@id='cphCuerpo_gvConteoServicios']//td[contains(text(),'" + destinos[i]
                        + "')]/following-sibling::td[contains(text(),'" + cantidadDestino[i] + "')]";

                WebElement celda = filas.get(i).findElements(By.tagName("td")).get(2);
                String contador = celda.getText();
                informacion.put("Tx acumuladas " + i, contador);

                WebElement cantMxTxRowTable = this.element(By.xpath(xpathComparar));
                if (cantMxTxRowTable == null) {
                    eventStatus = Reporter.MIC_FAIL;
                    Reporter.reportEvent(
                            eventStatus,
                            "Se esperaba que los datos de Administración de combos y empresa coincidieran.");
                    SettingsRun.exitTestIteration();
                }
            }

            eventStatus = Reporter.MIC_PASS;
            Reporter.reportEvent(eventStatus, "Los datos coinciden con Administración Combos.");

        } else {
            eventStatus = Reporter.MIC_FAIL;
            Reporter.reportEvent(
                    eventStatus,
                    "Se esperaba que la Cantidad Máxima Transacciones Empresa fuera igual en administración combos y en empresa.");
            SettingsRun.exitTestIteration();
        }
    }

    public String comboSeleccionado() throws Exception {
        this.consultarEmpresa();
        Evidence.save("Evidencia empresa", this);

        esperarElemento(tituloAdministracion, "No se encuentra el titulo de Administracion");

        WebElement comboServicios = this.element(By.id("cphCuerpo_ddlDEComboServicios"));
        Select select = new Select(comboServicios);

        return select.getFirstSelectedOption().getText();
    }

    /**
     * Compara el contador de transacciones con combos del después de la transacción
     */
    public void validarContadorPorTx() throws Exception {
        contadorActualizado = new int[5];
        Evidence.save("Evidencia Empresa", this);

        WebElement conteoTxTabla = this.element(tablaCantAcumulTx);
        List<WebElement> filas = conteoTxTabla.findElements(By.tagName("tr"));

        for (int i = 1; i < filas.size() - 1; i++) {
            try {
                WebElement celda = filas.get(i).findElements(By.tagName("td")).get(2);
                int contador = Integer.parseInt(celda.getText());
                contadorActualizado[i - 1] = contador;
            } catch (Exception e) {
                System.out.println(e.getMessage());
            }
        }

        Evidence.save("Evidencia empresa", this);
    }

    /**
     * Se mantiene almacenada la información sobre el conteo
     */
    public int[] getContadorAct() {
        return contadorActualizado;
    }
}
