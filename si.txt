import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import java.time.Duration;

public class TablaUtilsSimple {

    /**
     * Busca en la tabla el <a> cuyo texto sea igual a docNumber,
     * llama a This.click(...) y retorna true si lo hizo, false si no lo encontró o hubo error.
     */
    public static boolean clickDocumentSimple(WebDriver driver, String docNumber, int timeoutSeconds) {
        try {
            WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(timeoutSeconds));
            By locator = By.xpath(
                "//table[contains(@class,'table')]//tbody//a[normalize-space(text())='" + docNumber + "']"
            );

            WebElement link = wait.until(ExpectedConditions.elementToBeClickable(locator));
            This.click(link); // tu método de click existente
            return true;
        } catch (Exception e) {
            // Si no lo encuentra o falla el click, devolvemos false
            return false;
        }
    }
}
