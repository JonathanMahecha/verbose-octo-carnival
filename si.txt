	public void registrarLlave(String nombreLlaVinc) {
		int intentos = 0;
		int maxIntentos = 5;
		boolean llaveRegistro = buscarLlaveExistente(nombreLlaVinc);
		if (llaveRegistro) {
			waitElement(loc.bottonRegistro, 5);
			click(loc.bottonRegistro);
			Util.wait(4);
			while (intentos < maxIntentos) {
				if (isDisplayed(loc.sectionLoader)) {
					Util.wait(1);
				} else {
					break;
				}
				intentos++;
			}
		} else {

		}
	}
