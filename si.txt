package Actions;

import java.util.List;

import org.openqa.selenium.*;
import org.openqa.selenium.interactions.Actions;
import org.openqa.selenium.support.PageFactory;
import org.openqa.selenium.support.pagefactory.AjaxElementLocatorFactory;

import Locators.Locators_Registros_Breb_b;
import library.common.Util;
import library.core.BasePageWeb;
import library.reporting.Reporter;

public class Page_Registro_Breb_b extends BasePageWeb {

    private static final int TIMEOUT = 10;
    private static final int MAX_RETRIES = 15;

    private Locators_Registros_Breb_b loc;

    public Page_Registro_Breb_b(BasePageWeb parent) {
        super(parent);
        loc = new Locators_Registros_Breb_b();
        PageFactory.initElements(new AjaxElementLocatorFactory(getDriver(), 0), loc);
    }

    // ============================================================
    public void registrarLlave(String tipo, String nombre, String actividad, String ciudad,
                               String direccion, String complemento, String celular, String correo,
                               String info, String tipoCuenta, String numCuenta, String tipoLlave) throws Exception {

        if (tipo != null && tipo.contains("Personales")) {
            flujoPersonal(numCuenta, tipoLlave);
            return;
        }

        click(loc.bottonRegistro);

        if (llaveYaExiste(nombre)) {
            fail("Ya existe llave");
        }

        write(loc.inputLlave, nombre);

        seleccionarCuenta(tipoCuenta, numCuenta);

        seleccionarTipoLlave(tipoLlave);

        validar();
    }

    // ============================================================
    private void flujoPersonal(String numCuenta, String tipoLlave) {

        click(loc.buttonLlavePerson);

        seleccionarCuentaPorUltimos4(numCuenta);
        seleccionarRadio(tipoLlave);

        click(loc.buttonContinuar);
    }

    private void seleccionarCuenta(String tipo, String numero) {

        click(loc.barraCuentas);

        write(By.xpath("//input"), numero);

        if (tipo != null && tipo.toLowerCase().contains("ahorros")) {
            click(loc.botonAhorros);
        } else {
            click(loc.botonCorriente);
        }

        click(loc.botonPriCuenta);
        click(loc.botonAceptar);
        click(loc.botonContinuarRegistro);
    }

    private void seleccionarTipoLlave(String tipo) {

        By locator = By.xpath("//dav-text[contains(.,'" + tipo + "')]/ancestor::label//span");

        if (!isDisplayed(locator)) {
            fail("No existe tipo llave");
        }

        click(locator);
    }

    private void validar() {

        if (!isDisplayed(loc.registExtiText)) {
            fail("Registro falló");
        }

        Reporter.reportEvent(Reporter.MIC_PASS, "Registro OK");
    }

    // ============================================================
    public boolean llaveYaExiste(String nombre) {

        write(loc.searchBarLlav, nombre);
        click(loc.botonSearch);

        return isDisplayed(loc.tablaDatos);
    }

    private void seleccionarCuentaPorUltimos4(String numero) {

        String ult4 = numero.substring(numero.length() - 4);

        List<WebElement> filas = findElements(By.xpath("//*[@class='dav-cardResume']"));

        for (WebElement f : filas) {
            if (f.getText().contains(ult4)) {
                f.click();
                return;
            }
        }

        fail("No encontró cuenta");
    }

    private void seleccionarRadio(String texto) {

        List<WebElement> radios = findElements(By.xpath("//div[@role='radio']"));

        for (WebElement r : radios) {
            if (r.getText().contains(texto)) {
                r.click();
                return;
            }
        }

        fail("No encontró opción");
    }

    private void fail(String msg) {
        Reporter.reportEvent(Reporter.MIC_FAIL, msg);
        throw new RuntimeException(msg);
    }
}
