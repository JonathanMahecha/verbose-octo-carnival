public String closeActiveIntAlert() throws Exception {
    String msgResp = null;

    if (waitForVisibleAndEnabled(btnModalPopup, 10)) {
        msgResp = getMsgAlertIfExist("AlertaModal");
        Evidence.saveAllScreens("Alert", this);
        this.click(btnModalPopup);
        return msgResp;
    }

    if (waitForVisibleAndEnabled(cmpPopup, 10)) {
        msgResp = getMsgAlertIfExist("mensaje");
        Evidence.saveAllScreens("Alert", this);
        this.click(btnPopup);
        return msgResp;
    }

    if (waitForVisibleAndEnabled(AlertPopupAdver, 10)) {
        msgResp = getMsgAlertIfExist("mensaje");
        Evidence.saveAllScreens("Alert", this);
        this.click(btnAcetarAler);
        return msgResp;
    }

    if (waitForVisibleAndEnabled(btnPopup2, 10)) {
        msgResp = getMsgAlertIfExistxPath(Alermod);
        Evidence.saveAllScreens("Alert", this);
        this.click(btnPopup2);
        return msgResp;
    }

    if (waitForVisibleAndEnabled(Alermod2, 10)) {
        msgResp = getMsgAlertIfExistxPath(Alermod2);
        Evidence.saveAllScreens("Alert", this);

        if (this.element(btnPopupaceptar) != null) {
            this.click(btnPopupaceptar);
        } else if (this.element(btnPopupAvertaceptar) != null) {
            this.click(btnPopupAvertaceptar);
        }

        return msgResp;
    }

    return null;
}

private boolean waitForVisibleAndEnabled(By locator, int timeoutSeconds) throws Exception {
    int contador = 0;

    while (contador < timeoutSeconds) {
        if (this.element(locator) != null && this.isDisplayed(locator) && this.isEnabled(locator)) {
            return true;
        }

        Util.wait(1);
        contador++;
    }

    return false;
}
