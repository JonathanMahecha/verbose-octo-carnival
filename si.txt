public String ingresarDinamica(int numRetosFalla) throws Exception {

    String dato = obtenerEtiquetaToken();

    String[] arrOtps = DatosEmpresarial.getArrayToken();
    if (arrOtps.length == 0) {
        return MSG_ERROR_NO_OTP;
    }

    String[] arrTokIngresar = construirArrayTokens(arrOtps, numRetosFalla);

    boolean pideCvv = this.element(locCmCvv) != null;
    boolean pideCvvPenAp = this.element(locCmCvvPenAp) != null;

    String valorEstatico = SettingsRun.getTestData()
            .getParameter("Semilla / Valor Estático / Celular");

    String msgRta = null;
    int numIngresos = 0;

    for (int posArr = 0; posArr < arrTokIngresar.length; posArr++) {

        limpiarScriptToken();

        if (ingresarCampoSiExiste(locCmTokenOtpADEFPend, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpADEF, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpOrpa, valorEstatico)) {
            msgRta = "";
        }

        if (this.element(locCmTokenOtpInternacional) != null) {
            if (!this.isEnabled(locCmTokenOtpInternacional)) {
                break;
            }

            Util.wait(2);
            this.focus(locCmTokenOtpInternacional);
            Util.wait(3);

            msgRta = this.getActiveIntAlert();
            if (msgRta != null && !msgRta.isEmpty()) {
                return msgRta;
            }

            if (this.isEnabled(locCmTokenOtpInternacional)) {
                this.write(locCmTokenOtpInternacional, valorEstatico);
            }
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtp, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpPen, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpPen, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpModif, valorEstatico)) {
            msgRta = "";
        }

        if (ingresarCampoSiExiste(locCmTokenOtpElim, valorEstatico)) {
            msgRta = "";
        }

        if (pideCvv) {
            String cvv = SettingsRun.getTestData().getParameter("CVV-(código de seguridad TC)").trim();
            this.write(locCmCvv, cvv);
        } else if (pideCvvPenAp) {
            String cvv = SettingsRun.getTestData().getParameter("CVV-(código de seguridad TC)").trim();
            this.write(locCmCvvPenAp, cvv);
        }

        Util.wait(3);

        this.fechaHoraTx = new Date();
        Evidence.save("Ingreso" + dato + "(" + (posArr + 1) + ")", this);

        if (this.element(locCmTokenOtpInternacional) != null && this.isEnabled(locCmTokenOtpInternacional)) {
            Util.wait(2);
            BonotesTecla("TAB");
            BonotesTecla("ENTER");
            Evidence.save("Confirmación Clave", this);
            Util.wait(5);
        }

        boolean bloquearConfirmacion = this.isEnabled(locCmTokenOtpInternacional)
                || this.isEnabled(locCmTokenOtpOrpa)
                || this.isEnabled(locCmTokenOtpADEF)
                || this.isEnabled(locCmTokenOtpADEFPend);

        if (!bloquearConfirmacion) {
            String tipoPrueba = SettingsRun.getTestData().getParameter("Tipo prueba").trim();

            if (esTipoPruebaEspecial(tipoPrueba)) {
                confirmarTransaccionEspecial();
            } else {
                confirmarTransaccionComun();
            }
        }

        numIngresos++;

        boolean esInternacionalHabilitado = this.isEnabled(locCmTokenOtpInternacional);
        if (!esInternacionalHabilitado && !this.isEnabled(locCmTokenOtpOrpa)
                && !this.isEnabled(locCmTokenOtpADEF) && !this.isEnabled(locCmTokenOtpADEFPend)) {

            int conteo = 0;
            do {
                Util.wait(1);
                msgRta = this.getMsgAlertIfExist("lblMasterAlerta");

                if (msgRta == null) {
                    msgRta = this.getMsgAlertIfExist("AlertaModal");
                }
                if (msgRta == null) {
                    msgRta = this.getActiveIntAlert();
                }
                if (msgRta == null) {
                    msgRta = this.getMsgAlertIfExist("lblMasterTransaccionExitosaAlerta");
                }

                conteo++;
            } while ((msgRta == null || msgRta.contains(MSG_EXITO_ENV_OTP)) && conteo < 10);

            if (msgRta == null) {
                Evidence.saveFullPage("Aler", this);
                Reporter.write("Error en la aprobacion");
                this.pageLogin.CerrarSesionFront();
                SettingsRun.exitTestIteration();
            }

            Evidence.save("Aprobacion de transaccion hr", this);
        }

        String tipoPrueba = SettingsRun.getTestData().getParameter("Tipo prueba").trim();
        if (!tipoPrueba.equals("Tx Pend aprobación")) {
            Util.wait(4);
            if (this.isEnabled(locCmTokenOtpInternacional)) {
                int contador = 0;
                do {
                    Util.wait(1);
                    msgRta = this.getActiveIntAlert();
                    contador++;
                    if (contador >= 30) {
                        return msgRta;
                    }
                } while (msgRta == null || msgRta.contains(MSG_EXITO_ENV_OTP));
            }
        } else {
            if (this.isEnabled(locCmTokenOtpInternacional)) {
                int contador = 0;
                do {
                    Util.wait(1);
                    msgRta = this.getActiveIntAlert();
                    if (msgRta == null) {
                        msgRta = this.closeActiveIntAlert();
                    }
                    contador++;
                    if (contador >= 30) {
                        return msgRta;
                    }
                } while (msgRta == null || msgRta.contains(MSG_EXITO_ENV_OTP));
            }
        }
    }

    msgRta = this.getMsgAlertIfExist("lblMasterAlerta");
    if (msgRta == null) {
        msgRta = this.getMsgAlertIfExist("lblMasterTransaccionExitosaAlerta");
    }
    if (msgRta == null) {
        msgRta = this.getMsgAlertIfExist("AlertaModal");
    }

    if (msgRta != null && msgRta.contains(
            "Su código de aprobación ha expirado. De clic nuevamente en Aprobar Transacción para generar un código nuevo.")) {
        msgRta = this.aprobarTx2(0);
    }

    if (numRetosFalla != 0
            && numRetosFalla + 1 != numIngresos
            && !msgRta.contains("Su código de aprobación ha expirado. De clic nuevamente en Aprobar Transacción para generar un código nuevo.")) {
        msgRta += "\nPERO NO REALIZÁ LOS RETOS SOLICITADOS : [" + numRetosFalla + "]";
    }

    return msgRta;
}

private String obtenerEtiquetaToken() {
    String dato = "Token";
    if (DatosEmpresarial.TIPO_TOKEN.equals(DatosEmpresarial.TOKEN_OTP)) {
        dato = "Otp";
    }
    return dato;
}

private String[] construirArrayTokens(String[] arrOtps, int numRetosFalla) {
    if (numRetosFalla == 0) {
        return arrOtps;
    }

    String[] arrTokIngresar = new String[numRetosFalla + 1];
    int longTok = arrOtps[0].length();
    int lastNum = Integer.valueOf(Util.right(arrOtps[0], 1));

    for (int posArr = 0; posArr < numRetosFalla; posArr++) {
        lastNum++;
        arrTokIngresar[posArr] = Util.left(arrOtps[0], longTok - 1)
                + Util.right(String.valueOf(lastNum), 1);
    }

    arrTokIngresar[numRetosFalla] = arrOtps[0];
    return arrTokIngresar;
}

private boolean ingresarCampoSiExiste(By locator, String valor) throws Exception {
    if (this.element(locator) == null) {
        return false;
    }

    if (!this.isEnabled(locator)) {
        return false;
    }

    this.focus(locator);
    this.write(locator, valor);
    return true;
}

private void limpiarScriptToken() {
    this.getJse().executeScript(
            "var script = document.querySelector('script[src*=\"CSNYP.js\"]'); if (script) { script.parentNode.removeChild(script); }");
}

private boolean esTipoPruebaEspecial(String tipoPrueba) {
    return tipoPrueba.equals("Inscribir cuenta")
            || tipoPrueba.equals("Inscripción por archivo")
            || tipoPrueba.equals("Modificar cuenta")
            || tipoPrueba.equals("Eliminar cuenta")
            || tipoPrueba.equals("Consultar Inscripción")
            || tipoPrueba.equals("Autogestion de servicios");
}

private void confirmarTransaccionEspecial() throws Exception {
    String tipoPrueba = SettingsRun.getTestData().getParameter("Tipo prueba").trim();

    if (tipoPrueba.equals("Inscribir cuenta") || tipoPrueba.equals("Inscripción por archivo")
            || tipoPrueba.equals("Modificar cuenta") || tipoPrueba.equals("Eliminar cuenta")
            || tipoPrueba.equals("Consultar Inscripción") || tipoPrueba.equals("Autogestion de servicios")) {

        if (this.element(By.xpath("//input[@type='submit'][@value='Confirmar Transacción']")) != null) {
            this.element("//input[@type='submit'][@value='Confirmar Transacción']").click();
        } else if (this.element(By.xpath("//input[@type='submit'][@value='Confirmar Eliminación']")) != null
                || this.element(By.xpath("//*[@id='cphCuerpo_btnConfEliminar']")) != null) {
            this.clickButton("Confirmar Eliminación");
        } else if (this.element(By.xpath("//input[@type='submit'][@value='Confirmar Modificación']")) != null) {
            this.clickButton("Confirmar Modificación");
        } else if (this.element(By.id("btnConfirmar")) != null && this.isDisplayed(By.id("btnConfirmar"))) {
            this.clickButton("Confirmar Transacción");
        }
    }
}

private void confirmarTransaccionComun() throws Exception {
    Util.wait(1);

    if (this.isEnabled(this.elementButton("Confirmar Transacción"))) {
        this.element("//*[@value='Confirmar Transacción']").click();
    }

    if (this.isDisplayed(By.xpath("//*[@value='Confirmar']"))) {
        this.element(By.xpath("//*[@value='Confirmar']")).click();
    }

    if (this.isEnabled(this.elementButton("Actualizar Datos"))) {
        this.mouseOver(By.xpath("(//*[@id='cphCuerpo_TbAprobacion']//td)[6]"));
        this.mouseClick();
        Util.wait(5);
        BonotesTecla("TAB");
        this.clickButton("Actualizar Datos");
    }
}
