    public void doingConfigurations() throws Exception {
        boolean generarAleatorios = "SI".equalsIgnoreCase(SettingsRun.getGlobalData("prop.DataMasiva"));
        if (!generarAleatorios) {
            SettingsRun.getTestData().validarParameters(SettingsRun.ARRAY_DATA_PARAMS);
        }
    }

    /**
     * Punto de entrada principal: orquesta la carga de datos, validaciones,
     * navegación y exportación de resultados.
     */
    public void doTest() throws Exception {
        boolean usarAleatorios = "SI".equalsIgnoreCase(SettingsRun.getGlobalData("prop.DataMasiva"));

        if (!usarAleatorios) {
            SettingsRun.getTestData().validarParameters(SettingsRun.ARRAY_DATA_PARAMS);
        }

        DataDriven td = SettingsRun.getTestData();
        folderDestino = SettingsRun.RESULT_DIR + File.separator;
        if (!Util.directoryExist(folderDestino)) {
            new File(folderDestino).mkdirs();
        }

        if (usarAleatorios) {
            cargarDatosAleatorios();
        } else {
            cargarDatosDesdeArchivo(td);
        }

        validarNumeroIdentificacion();
        validarRangosPorTipoId();

        Reporter.write("Usuario: " + id + "\n Contraseña:" + contraseña);
        Reporter.write("*** Transando en portal c360");

        iniciarSesion();
        navegarAModuloInicial();
        verificarYRegenerarIdSiExiste();

        if (tipoCliente.contains("Persona Juridica")) {
            procesarPersonaJuridica();
        } else {
            procesarPersonaNatural();
        }

        exportarFilaExcelTodosDatosConHeaderRojo("DatosGenerados.xlsx", HEADERS, construirValoresExport());
    }

    /** Libera los recursos del browser. */
    public void close() {
        if (pageInicio != null) {
            pageInicio.closeCurrentBrowser();
        }
        if (c360 != null) {
            c360.closeCurrentBrowser();
        }
    }

    /** Alias de {@link #close()} para cumplir con la interfaz Controller. */
    public void destroy() {
        close();
    }

    private void validarNumeroIdentificacion() {
        try {
            Long.parseLong(numIdentificacion);
        } catch (NumberFormatException e) {
            Reporter.reportEvent(Reporter.MIC_NOEXEC,
                "[ERROR DATA] Número de identificación no numérico: " + numIdentificacion);
            SettingsRun.exitTestIteration();
        }
    }

    private void validarRangosPorTipoId() {
        long numId = Long.parseLong(numIdentificacion);

        switch (tipoIdentificacion) {
            case "NIT EXTRANJERIA":
            case "NIT MENORES":
                if (numId >= 6_000_000_000L && numId <= 7_999_999_999L) {
                    Reporter.write("ok");
                } else {
                    Reporter.reportEvent(Reporter.MIC_NOEXEC,
                        "[ERROR DATA] Número de identificación inválido para Nit Extranjeria y Nit Menores,"
                        + " los valores validos son: igual o mayor a 6,000,000,000 / menor o igual a 7,999,999,999");
                    SettingsRun.exitTestIteration();
                }
                break;
            case "REGISTRO CIVIL DE NACIMIENTO":
                if (numId <= 1_999_999_999L) {
                    Reporter.write("ok");
                } else {
                    Reporter.reportEvent(Reporter.MIC_NOEXEC,
                        "[ERROR DATA] Número de identificación inválido para Registro Civil de Nacimiento,"
                        + " los valores validos son: igual o inferior a 1,999,999,999");
                    SettingsRun.exitTestIteration();
                }
                break;
            case "PERMISO PROTECCION TEMPORAL":
                if (numId >= 100_000L && numId <= 99_999_999L) {
                    Reporter.write("ok");
                } else {
                    Reporter.reportEvent(Reporter.MIC_NOEXEC,
                        "[ERROR DATA] Número de identificación inválido para Permiso Protección Temporal,"
                        + " los valores validos son: igual o mayor a 100.000 y menor o igual a 99.999.999");
                    SettingsRun.exitTestIteration();
                }
                break;
            default:
                break;
        }
    }

    private void iniciarSesion() {
        try {
            this.c360 = new PageLogin(BasePageWeb.CHROME);
            this.c360.maximizeBrowser();
            this.pageInicio = new PageInicioC360(c360);
            pageInicio.refresh();
            this.c360.login360(id, contraseña);

            do {
                Util.wait(3);
                pageInicio.reporteAlertas();
            } while (pageInicio.element(locBienvenidoMainPage) == null);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    private void navegarAModuloInicial() {
        if (tipoCliente.contains("Persona Natural")) {
            Util.wait(3);
            pageInicio.irAModulo(PageInicioC360.MOD_PERSONAS);

        } else if (tipoCliente.contains("Persona Juridica")) {
            if (tipoIdentificacion.contains("NIT")) {
                numIdentificacion = numIdentificacion + digitoVerificacion;
            }
            pageInicio.irAModulo(PageInicioC360.MOD_EMPRESAS);
        }
    }

    private void verificarYRegenerarIdSiExiste() throws Exception {
        String existe = pageInicio.buscarExistenciaId(numIdentificacion, tipoCliente);
        if (existe == null) {
            return;
        }

        do {
            Reporter.reportEvent(Reporter.MIC_INFO,
                "Este número de Identificacion ya esta registrado en Cliente 360: " + numIdentificacion);
            numIdentificacion  = generarIdentifiacion(tipoIdentificacion, numIdentificacion);
            digitoVerificacion = digitoVerificacion(numIdentificacion);

            pageInicio.irAModulo(PageInicioC360.MOD_PAGINA_INICIAL);

            if (tipoCliente.contains("Persona Natural")) {
                Util.wait(3);
                pageInicio.irAModulo(PageInicioC360.MOD_PERSONAS);
            } else if (tipoCliente.contains("Persona Juridica")) {
                pageInicio.irAModulo(PageInicioC360.MOD_EMPRESAS);
            }

            existe = pageInicio.buscarExistenciaId(numIdentificacion, tipoCliente);
        } while (existe != null);
    }

                tipoContrato, detallesActEconomica, nitEmpresa, cargoOcupacion, desIngresos, nombreEmpresa);
            pageInicio.guardar();
        }

            pageInicio.irAModulosPrincipales(locInformacionFinanciera);
            pageInicio.agregarNuevo(locBtAgregarPasivos);
            pageInicio.setPasivos(tipoPasivos, valorPasivos, descripcionPasivos);
            pageInicio.agregarNuevo(locBtAgregarActivos);
            pageInicio.reporteAlertas();
        }

        if (!tipoActivos.isEmpty()) {
            Util.wait(1);
            pageInicio.setActivos(tipoActivos, valorActivos, descripcionActivos);
            pageInicio.guardar();
        }

        this.c360.tomarCaptura("Datos información financiera");
        Reporter.reportEvent(Reporter.MIC_INFO, "Completado: Datos información financiera.");
        Reporter.reportEvent(Reporter.MIC_INFO,
            "Registro correspondiente a Tipo de Identifiacion:" + tipoIdentificacion
            + " Número de Identificación: " + numIdentificacion);

        pageInicio.guardar();
        Reporter.reportEvent(Reporter.MIC_PASS, "REGISTRO REALIZADO");
        pageInicio.closeAllBrowsers();
    }

    private void exportarFilaExcelTodosDatosConHeaderRojo(String fileName, String[] headers, String[] values) {
        Path path = Paths.get(folderDestino + fileName).toAbsolutePath();
        Workbook workbook = null;
        Sheet sheet = null;

        try {
            if (Files.exists(path)) {
                try (FileInputStream fis = new FileInputStream(path.toFile())) {
                    workbook = new XSSFWorkbook(fis);
                }
                sheet = workbook.getSheet("DatosGenerados");
                if (sheet == null) {
                    sheet = workbook.createSheet("DatosGenerados");
                    createHeaderWithRedBackground(workbook, sheet, headers);
                }
            } else {
                workbook = new XSSFWorkbook();
                sheet = workbook.createSheet("DatosGenerados");
                createHeaderWithRedBackground(workbook, sheet, headers);
            }

            int appendRowIdx = sheet.getLastRowNum() + 1;
            if (sheet.getPhysicalNumberOfRows() == 0) {
                createHeaderWithRedBackground(workbook, sheet, headers);
                appendRowIdx = 1;
            }

            Row row = sheet.createRow(appendRowIdx);
            for (int i = 0; i < values.length; i++) {
                Cell cell = row.createCell(i);
                cell.setCellValue(values[i] == null ? "" : values[i]);
            }

            for (int i = 0; i < headers.length; i++) {
                sheet.autoSizeColumn(i);
            }

            try (FileOutputStream fos = new FileOutputStream(path.toFile())) {
                workbook.write(fos);
            }

            Reporter.reportEvent(Reporter.MIC_INFO, "Excel actualizado: " + path);

        } catch (Exception e) {
            Reporter.reportEvent(Reporter.MIC_FAIL, "Error al escribir Excel: " + e.getMessage());
        } finally {
            if (workbook != null) {
                try { workbook.close(); } catch (Exception ignored) { }
            }
        }
    }

    private void createHeaderWithRedBackground(Workbook workbook, Sheet sheet, String[] headers) {
        CellStyle style = workbook.createCellStyle();
        Font font = workbook.createFont();
        font.setBold(true);
        font.setColor(IndexedColors.WHITE.getIndex());
        style.setFont(font);
        style.setFillForegroundColor(IndexedColors.RED.getIndex());
        style.setFillPattern(FillPatternType.SOLID_FOREGROUND);
        style.setAlignment(HorizontalAlignment.CENTER);
        style.setVerticalAlignment(VerticalAlignment.CENTER);

        Row headerRow = sheet.createRow(0);
        for (int i = 0; i < headers.length; i++) {
            Cell cell = headerRow.createCell(i);
            cell.setCellValue(headers[i]);
            cell.setCellStyle(style);
        }
    }

    public String digitoVerificacion(String nit) {
        int[] vpri = new int[16];
        vpri[1]  = 3;  vpri[2]  = 7;  vpri[3]  = 13; vpri[4]  = 17;
        vpri[5]  = 19; vpri[6]  = 23; vpri[7]  = 29; vpri[8]  = 37;
        vpri[9]  = 41; vpri[10] = 43; vpri[11] = 47; vpri[12] = 53;
        vpri[13] = 59; vpri[14] = 67; vpri[15] = 71;

        int x = 0;
        int z = nit.length();
        for (int i = 0; i < z; i++) {
            int y = Integer.parseInt(String.valueOf(nit.charAt(i)));
            x += y * vpri[z - i];
        }

        int y = x % 11;
        return Integer.toString(y > 1 ? 11 - y : y);
    }

    public String generarIdentifiacion(String tipoIdentificacion, String numIdentificacion) throws Exception {
        long randomId;

        switch (tipoIdentificacion) {
            case "CEDULA DE CIUDADANIA": {
                String prefix = RAND.nextBoolean() ? "10" : "11";
                int resto = RAND.nextInt(100_000_000);
                return prefix + String.format("%08d", resto);
            }
            case "CEDULA DE EXTRANJERIA":
            case "CÓDIGO FISCAL TRIBUTARIO":
            case "CARNET DIPLOMATICO":
            case "SOC EXTRAN SIN NIT EN COLOMBIA":
            case "TRJ. SEGURO SOCIAL EXTRANJERO":
            case "PASAPORTE":
                randomId = Math.abs(RAND.nextLong() % (2_999_999_999L - 1_000_000_000L + 1)) + 1_000_000_000L;
                numIdentificacion = removerUltimoCaracter(Long.toString(randomId));
                Reporter.write("Identificacion Generada: " + tipoIdentificacion + " " + numIdentificacion);
                SettingsRun.getTestData().setParameter("Número de Identificación", numIdentificacion);
                break;

            case "NIT":
            case "NIT PERSONA NATURAL":
                randomId = Math.abs(RAND.nextLong() % (999_999_999L - 800_000_000L + 1)) + 800_000_000L;
                numIdentificacion = Long.toString(randomId);
                Reporter.write("Nit Menores/Nit Extranjeria " + numIdentificacion);
                SettingsRun.getTestData().setParameter("Número de Identificación", numIdentificacion);
                break;

            case "NIT EXTRANJERIA":
            case "NIT MENORES":
                randomId = Math.abs(RAND.nextLong() % 1_999_999_999L) + 6_000_000_000L;
                numIdentificacion = Long.toString(randomId);
                Reporter.write("Nit Menores/Nit Extranjeria " + numIdentificacion);
                SettingsRun.getTestData().setParameter("Número de Identificación", numIdentificacion);
                break;

            case "REGISTRO CIVIL DE NACIMIENTO":
                randomId = Math.abs(RAND.nextLong() % (1_999_999_999L + 1));
                numIdentificacion = Long.toString(randomId);
                Reporter.write("Permiso de proteccion temporal " + numIdentificacion);
                SettingsRun.getTestData().setParameter("Número de Identificación", numIdentificacion);
                break;

            case "PERMISO PROTECCION TEMPORAL":
                randomId = Math.abs(RAND.nextLong() % (99_999_999 - 100_000 + 1)) + 100_000;
                numIdentificacion = Long.toString(randomId);
                Reporter.write("Permiso de proteccion temporal " + numIdentificacion);
                SettingsRun.getTestData().setParameter("Número de Identificación", numIdentificacion);
                break;

            default:
                break;
        }

        return numIdentificacion;
    }

    public static LocalDate parseFechaFlexible(String fecha) {
        if (fecha == null || fecha.trim().isEmpty()) {
            throw new IllegalArgumentException("Fecha vacía o nula");
        }

        String f = fecha.trim();
        List<DateTimeFormatter> formatters = Arrays.asList(
            DateTimeFormatter.ofPattern("d/M/yyyy"),
            DateTimeFormatter.ofPattern("d-M-yyyy"),
            DateTimeFormatter.ofPattern("dd/MM/yyyy"),
            DateTimeFormatter.ofPattern("dd-MM-yyyy"),
            DateTimeFormatter.ofPattern("d/MM/yyyy"),
            DateTimeFormatter.ofPattern("d-MM-yyyy"),
            DateTimeFormatter.ofPattern("yyyy-MM-dd"),
            DateTimeFormatter.ofPattern("yyyy/M/d")
        );

        for (DateTimeFormatter fmt : formatters) {
            try {
                return LocalDate.parse(f, fmt);
            } catch (DateTimeParseException ignored) { }
        }

        throw new IllegalArgumentException("Formato de fecha no válido: " + fecha);
    }

    public static String getRandomDatos(String[] datos) {
        if (datos == null || datos.length == 0) {
            throw new IllegalArgumentException("El arreglo esta null");
        }
        return datos[RAND.nextInt(datos.length)];
    }

    private static String generarTelefonoLocal() {
        return String.format("72%05d", RAND.nextInt(100_000));
    }

    private static String generarCelular() {
        String[] prefijos = { "320", "322" };
        String prefix = prefijos[RAND.nextInt(prefijos.length)];
        return String.format("%s%07d", prefix, RAND.nextInt(10_000_000));
    }

    private static String generarDireccion() {
        int a = RAND.nextInt(200) + 1;
        int b = RAND.nextInt(100);
        int c = RAND.nextInt(100);
        return String.format("KR %d %02d %02d", a, b, c);
    }

    private static String[] generarFechasNacimientoYExpedicion() {
        LocalDate hoy      = LocalDate.now();
        int edad           = RAND.nextInt(46) + 20;
        LocalDate nacimiento = hoy.minusYears(edad);
        LocalDate minExp   = nacimiento.plusYears(19);
        LocalDate maxExp   = hoy;

        long daysRange = java.time.temporal.ChronoUnit.DAYS.between(minExp, maxExp);
        if (daysRange < 0) {
            minExp = hoy.minusYears(1);
            daysRange = java.time.temporal.ChronoUnit.DAYS.between(minExp, maxExp);
        }
        int addDays = (daysRange > 0) ? RAND.nextInt((int) daysRange + 1) : 0;
        LocalDate expedicion = minExp.plusDays(addDays);

        DateTimeFormatter fmt = DateTimeFormatter.ofPattern("dd-MM-yyyy");
        return new String[] { nacimiento.format(fmt), expedicion.format(fmt) };
    }

    private static String removerUltimoCaracter(String input) {
        if (input != null && input.length() > 3) {
            return input.substring(0, input.length() - 3);
        }
        Reporter.write("NO se pudo remover el ultimo caracter.");
        return input;
    }

    private static String getParam(DataDriven td, String key) {
        try {
            String v = td.getParameter(key);
            return v == null ? "" : v;
        } catch (Exception e) {
            return "";
        }
    }

    private static String getParamOrDefault(DataDriven td, String key, String defaultVal) {
        String v = getParam(td, key);
        return v.isEmpty() ? defaultVal : v;
    }

    private static String telefoneOrDefault(String tel) {
        return (tel == null || tel.trim().isEmpty()) ? "0000000" : tel;
    }
