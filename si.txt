	public String getCodigoXNombreContained(String nombre) {

		String codigo = null;
		String nombreBuscar = Util.removeAccents(nombre.trim().toUpperCase());
		if (catalogoXNombre.containsKey(nombreBuscar))
			codigo = catalogoXNombre.get(nombreBuscar);
		else {
			for (String keyNombre : catalogoXNombre.keySet()) {
				if (keyNombre.contains(nombreBuscar) || nombreBuscar.contains(keyNombre)) {
					codigo = catalogoXNombre.get(keyNombre);
					break; // PARA QUE TERMINE EL CICLO
				}
			}
		}
		return codigo;
	}
