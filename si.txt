	private boolean cambiarAVentanaEmergente() throws Exception {
		Util.wait(5);

		Set<String> handles = loginFrontPyme.getDriver().getWindowHandles();
		List<String> ventanas = new ArrayList<>(handles);

		if (ventanas.size() < 2) {
			reportFail("No se abrio la segunda pestaña");
			SettingsRun.exitTestIteration();
			return false;
		}

	
		
		
		if (ventanas.size() > 3) {
			for (String ventana : ventanas) {
				Reporter.reportEvent(Reporter.MIC_INFO, ventana);
				String ventana1 = ventana;
				loginFrontPyme.getDriver().switchTo().window(ventana1);
				WebElement empresa = loginFrontPyme.element(By.xpath("//select[@id='dropMasterEmpresa']"));
				if (!loginFrontPyme.isDisplayed(empresa)) {
					loginFrontPyme.getDriver().close();
				}
				break;
			}
		}
		
		String primeraVentana = ventanas.get(0);
		String segundaVentana = ventanas.get(1);

		Reporter.write("Primera venta: " + primeraVentana + " || Segunda venta: " + segundaVentana);

		loginFrontPyme.getDriver().switchTo().window(primeraVentana);
		Util.wait(2);
		loginFrontPyme.getDriver().close();
		loginFrontPyme.getDriver().switchTo().window(segundaVentana);

		return true;
	}
